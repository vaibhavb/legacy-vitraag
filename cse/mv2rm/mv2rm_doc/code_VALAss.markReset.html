let markReset s =
  if (search_sym_tab s) then begin
    match (Hashtbl.find symTab s) with
      Symb(a, b, None) ->
        Hashtbl.replace symTab s (
        Symb(a,b,ResetAtom(
             Controls(!refoutList),
             Awaits(!refinList),
             Relations(!relations)
               )))
    |Symb(a,b,LatchAtom(
             LControls(s),
             RControls(p),
             Reads(input),
             Awaits(q),
             Relations(r))) ->
                       Hashtbl.replace symTab s (
        Symb(a,b,LatchAtom(
             LControls(s),
             RControls(!refoutList),
             Reads(input),
             Awaits(!refinList),
             Relations(!relations)
            )))
    | _ ->  raise (Failure("Already Controlled :"^s))
  end
