  let ft = ref true in
  let n = ref ""  in
  let f c =

    match str with
      "$" ->  if (!ft) then (n := ((!n)^ "a&"); ft:=false )
      else n := ((!n)^ "&" )
    | "_" -> if (!ft) then (n := ((!n)^ "a&&"); ft:=false )
    else n := ((!n)^ "&&" )
    | "<" -> if (!ft) then (n := ((!n)^ "a&&&"); ft:=false )
    else n := ((!n)^ "&&&" )
    | ">" -> if (!ft) then (n := ((!n)^ "a&&&&"); ft:=false )
    else n := ((!n)^ "&&&&" )
    | "?" -> if (!ft) then (n := ((!n)^ "a&&&&&"); ft:=false )
    else n := ((!n)^ "&&&&&" )
    | "@" -> if (!ft) then (n := ((!n)^ "a&&&&&&"); ft:=false )
    else n := ((!n)^ "&&&&&&" )

    else n := ((!n) ^ str )
  in
