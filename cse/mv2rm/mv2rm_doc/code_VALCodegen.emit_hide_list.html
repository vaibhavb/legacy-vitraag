let emit_hide_list name varList =
  begin
  vemit ( name ^ " := hide " );
  vemit (List.hd varList);
  try 
    let listMinusFst = List.tl varList in
    List.iter (fun var ->
      vemit( ",");
      vemit (var);
      ) listMinusFst;
    vemit ( " in b_" ^ name ^" endhide\n");
  with Not_found ->
    vemit ( " in b_" ^ name ^" endhide\n");
  end
