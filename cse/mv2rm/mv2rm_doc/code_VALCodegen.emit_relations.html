let emit_relations refoutList refinList relations =
  let relemit s = 
    match s with
      [] -> ()
    | hd::tl  -> 
        let is_first = ref true in 
        let vemit2out a b = 
          match b with
          | "0" ->if (!is_first) then (vemit (a^"':= false "); 
                                       is_first:=false )
          else (vemit ("; "^a^"':= false"))
          | "1" ->if (!is_first) then (vemit (a^"':=true "); 
                                       is_first:=false )
          else (vemit ("; "^a^"':= true"))
          | "hyp" ->if (!is_first) then (vemit (a^"':= nondet "); 
                                       is_first:=false )
          else (vemit ("; "^a^"':= nondet"))
          | _ ->if (!is_first) then (vemit (a^"':="^b^" "); 
                                     is_first:=false )
          else (vemit ("; "^a^"':="^b^" "))
        in
        if (hd=".def") then begin
          vemit("\t [] default ->");
          let outList = 
            List.rev (split_list_from_start (List.rev s)
                        (List.length (refoutList)) ) in 
          is_first := true;
          List.iter2 vemit2out (refoutList) outList;
          vemit "\n"
        end
        else
          begin
            vemit "\t [] ";
            let vemit2in a b = begin 
              match b with
                "hyp" -> 
                  if (!is_first) then (vemit ("true "); is_first:=false )
                  else (vemit ("& true "))
              | "0" ->if (!is_first) then (vemit ("~"^a^"' "); 
                                           is_first:=false )
              else (vemit ("& ~"^a^"' "))
              | "1" ->if (!is_first) then (vemit (a^"' "); 
                                           is_first:=false )
              else (vemit ("& "^a^"' "))
              | _ ->
                  if (!is_first) then (vemit (a^"'="^b^" "); is_first:=false )
                  else (vemit ("& "^a^"'="^b^" "))
            end in         
            begin
              try 
                if (refinList <> []) then begin
                  let inList = 
                    split_list_from_start s (List.length (refinList)) in
                  is_first := true; 
                  List.iter2 vemit2in (refinList) inList;
                end
                else begin vemit "true " end;
                vemit ("->");
                is_first := true;
                let outList = 
                  List.rev (split_list_from_start (List.rev s)
                            (List.length (refoutList)) ) in 
                List.iter2 vemit2out (refoutList) outList
              with Invalid_argument t ->
                vemit (t ^ " " ^ string_of_int (List.length s) ^ " " ^
                       string_of_int (List.length ((refoutList))))
            end;
            vemit "\n"
          end
  in
  if (relations <> []) then begin 
    List.iter relemit relations
  end
